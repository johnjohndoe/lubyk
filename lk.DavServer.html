local function makeResponse(self, resource, prop_list) local prop = {xml='prop'} for _, k in ipairs(prop_list) do k = k.xml local v = resource:davProperty(k) if v then if k == 'creationdate' then v = os.date(DATE_FORMAT_ISO_8601, v) elseif k == 'getlastmodified' then -- This format is a stupid: it renders differently depending on the -- locale: this is why we need setlocale('C') v = os.date(DATE_FORMAT_HTTP_DATE, v) end end table.insert(prop, {xml = k, v or ''}) end return { xml = 'response', {xml = 'href', self.href_base .. resource:davProperty('url')}, {xml = 'propstat', prop, {xml = 'status', "HTTP/1.1 200 OK"}, } } end
